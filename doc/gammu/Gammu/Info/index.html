type charge_state =
| BatteryPowered

(** Powered from battery *)

| BatteryConnected

(** Powered from AC, battery connected *)

| BatteryCharging

(** Powered from AC, battery is charging *)

| BatteryNotConnected

(** Powered from AC, no battery *)

| BatteryFull

(** Powered from AC, battery is fully charged *)

| PowerFault

(** Power failure *)
